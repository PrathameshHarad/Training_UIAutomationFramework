pipeline {
    agent any

    tools {
        dotnet 'dotnet6' // Make sure this matches the name of your .NET installation in Jenkins
    }

    stages {
        stage('Restore') {
            steps {
                sh 'dotnet restore'
            }
        }

        stage('Build') {
            steps {
                sh 'dotnet build --configuration Release'
            }
        }

        stage('Run UI Tests') {
            steps {
                sh 'dotnet test --configuration Release --logger:"trx;LogFileName=test_results.trx"'
            }
        }
    }
}
